51  /**
52   * <p>Implementation of the <code>docker-crac</code> packaging.</p>
53   * <p><strong>WARNING</strong>: this goal is not intended to be executed directly. Instead, specify the packaging type
54   * using the <code>packaging</code> property, eg:</p>
55   *
56   * <pre>mvn package -Dpackaging=docker-crac</pre>
57   * <p>
58   * This is a two stage process. First a docker image is built that runs the application under a CRaC enabled JDK. Then
59   * the application is warmed up via a shell script. And then a checkpoint is taken via a signal using jcmd.
60   * <p>
61   * The second stage takes this checkpoint, and creates the final image containing it plus a run script which passes the
62   * correct flags to the CRaC enabled JDK.
63   *
64   * @author Tim Yates
65   * @since 3.5.0
66   */
67  @Experimental
68  @Mojo(name = DockerCracMojo.DOCKER_CRAC_PACKAGING, requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME)
69  public class DockerCracMojo extends AbstractDockerMojo {
70  
71      public static final String DOCKER_CRAC_PACKAGING = "docker-crac";
72      public static final String CHECKPOINT_SCRIPT_NAME = "checkpoint.sh";
73      public static final String WARMUP_SCRIPT_NAME = "warmup.sh";
74      public static final String RUN_SCRIPT_NAME = "run.sh";
75      public static final String DEFAULT_READINESS_COMMAND = "curl --output /dev/null --silent --head http://localhost:8080";
76      public static final String CRAC_READINESS_PROPERTY = "crac.readiness";
77      public static final String DEFAULT_CRAC_CHECKPOINT_TIMEOUT = "60";
78      public static final String CRAC_CHECKPOINT_NETWORK_PROPERTY = "crac.checkpoint.network";
79      public static final String CRAC_CHECKPOINT_TIMEOUT_PROPERTY = "crac.checkpoint.timeout";
80      public static final String DEFAULT_BASE_IMAGE = "ubuntu:22.04";
81  
82      private static final EnumSet<PosixFilePermission> POSIX_FILE_PERMISSIONS = EnumSet.of(
83              PosixFilePermission.OWNER_READ, PosixFilePermission.OWNER_WRITE, PosixFilePermission.OWNER_EXECUTE,
84              PosixFilePermission.GROUP_READ, PosixFilePermission.GROUP_WRITE, PosixFilePermission.GROUP_EXECUTE,
85              PosixFilePermission.OTHERS_READ, PosixFilePermission.OTHERS_EXECUTE
86      );
87      private final MavenReaderFilter mavenReaderFilter;
88  
89      @Parameter(property = DockerCracMojo.CRAC_READINESS_PROPERTY, defaultValue = DockerCracMojo.DEFAULT_READINESS_COMMAND)
90      private String readinessCommand;
91  
92      @Parameter(property = DockerCracMojo.CRAC_CHECKPOINT_TIMEOUT_PROPERTY, defaultValue = DockerCracMojo.DEFAULT_CRAC_CHECKPOINT_TIMEOUT)
93      private Integer checkpointTimeoutSeconds;
94  
95      @Parameter(property = DockerCracMojo.CRAC_CHECKPOINT_NETWORK_PROPERTY)
96      private String checkpointNetworkName;
97  
98      @SuppressWarnings("CdiInjectionPointsInspection")
99      @Inject
100     public DockerCracMojo(
101             MavenProject mavenProject,
102             JibConfigurationService jibConfigurationService,
103             ApplicationConfigurationService applicationConfigurationService,
104             DockerService dockerService,
105             MavenReaderFilter mavenReaderFilter
106     ) {
107         super(mavenProject, jibConfigurationService, applicationConfigurationService, dockerService);
108         this.mavenReaderFilter = mavenReaderFilter;
109     }
110 
111     @Override
112     public void execute() throws MojoExecutionException {
113         try {
114             copyDependencies();
115 
116             MicronautRuntime runtime = MicronautRuntime.valueOf(micronautRuntime.toUpperCase());
117 
118             switch (runtime.getBuildStrategy()) {
119                 case LAMBDA:
120                     throw new MojoExecutionException("Lambda Functions are currently unsupported");
121 
122                 case ORACLE_FUNCTION:
123                     throw new MojoExecutionException("Oracle Functions are currently unsupported");
124 
125                 case DEFAULT:
126                 default:
127                     buildDockerCrac();
128                     break;
129             }
130         } catch (InvalidImageReferenceException iire) {
131             String message = "Invalid image reference "
132                     + iire.getInvalidReference()
133                     + ", perhaps you should check that the reference is formatted correctly according to " +
134                     "https://docs.docker.com/engine/reference/commandline/tag/#extended-description" +
135                     "\nFor example, slash-separated name components cannot have uppercase letters";
136             throw new MojoExecutionException(message);
137         } catch (IOException | IllegalArgumentException | MavenFilteringException e) {
138             throw new MojoExecutionException(e.getMessage(), e);
139         }
140     }
141 
142     private void buildDockerCrac() throws IOException, InvalidImageReferenceException, MavenFilteringException {
143         String checkpointImage = buildCheckpointDockerfile();
144         getLog().info("CRaC Checkpoint image: " + checkpointImage);
145         File checkpointDir = new File(mavenProject.getBuild().getDirectory(), "cr");
146         // We need to make this folder first, or else Docker on linux will make it as root and break clean on CI
147         checkpointDir.mkdirs();
148         dockerService.runPrivilegedImageAndWait(
149                 checkpointImage,
150                 checkpointTimeoutSeconds,
151                 checkpointNetworkName,
152                 checkpointDir.getAbsolutePath() + ":/home/app/cr"
153         );
154         buildFinalDockerfile(checkpointImage);
155     }
156 
157     private String buildCheckpointDockerfile() throws IOException, MavenFilteringException {
158         String name = mavenProject.getArtifactId() + "-crac-checkpoint";
159         Set<String> checkpointTags = Collections.singleton(name);
160         copyScripts(CHECKPOINT_SCRIPT_NAME, WARMUP_SCRIPT_NAME, RUN_SCRIPT_NAME);
161         File dockerfile = dockerService.loadDockerfileAsResource(DockerfileMojo.DOCKERFILE_CRAC_CHECKPOINT);
162         BuildImageCmd buildImageCmd = dockerService.buildImageCmd()
163                 .withDockerfile(dockerfile)
164                 .withBuildArg("BASE_IMAGE", getFromImage().orElse(DEFAULT_BASE_IMAGE))
165                 .withTags(checkpointTags);
166         dockerService.buildImage(buildImageCmd);
167         return name;
168     }
169 
170     private void buildFinalDockerfile(String checkpointContainerId) throws IOException, InvalidImageReferenceException, MavenFilteringException {
171         Set<String> tags = getTags();
172         for (String tag : tags) {
173             ImageReference.parse(tag);
174         }
175         copyScripts(RUN_SCRIPT_NAME);
176         File dockerfile = dockerService.loadDockerfileAsResource(DockerfileMojo.DOCKERFILE_CRAC);
177         BuildImageCmd buildImageCmd = dockerService.buildImageCmd()
178                 .withDockerfile(dockerfile)
179                 .withBuildArg("BASE_IMAGE", getFromImage().orElse(DEFAULT_BASE_IMAGE))
180                 .withBuildArg("CHECKPOINT_IMAGE", checkpointContainerId)
181                 .withTags(getTags());
182         dockerService.buildImage(buildImageCmd);
183     }
184 
185     private Properties replacementProperties(String readinessCommand, String mainClass) {
186         Properties properties = new Properties();
187         properties.setProperty("READINESS", readinessCommand);
188         properties.setProperty("MAINCLASS", mainClass);
189         return properties;
190     }
191 
192     private void copyScripts(String... scriptNames) throws IOException, MavenFilteringException {
193         File target = new File(mavenProject.getBuild().getDirectory(), "scripts");
194         if (!target.exists()) {
195             target.mkdirs();
196         }
197         processScripts(target, replacementProperties(readinessCommand, mainClass), scriptNames);
198     }
199 
200     private void processScripts(File target, Properties replacements, String... scriptNames) throws IOException, MavenFilteringException {
201         for (String script : scriptNames) {
202             File localOverride = new File(mavenProject.getBasedir(), script);
203             InputStream resourceStream = DockerCracMojo.class.getResourceAsStream("/cracScripts/" + script);
204             Reader resourceReader = resourceStream == null ? null : new InputStreamReader(resourceStream);
205             try (Reader reader = localOverride.exists() ? Files.newBufferedReader(localOverride.toPath()) : resourceReader) {
206                 if (reader == null) {
207                     throw new IOException("Could not find script " + script);
208                 }
209                 MavenReaderFilterRequest req = new MavenReaderFilterRequest();
210                 req.setFrom(reader);
211                 req.setFiltering(true);
212                 req.setAdditionalProperties(replacements);
213                 Path outputPath = target.toPath().resolve(script);
214                 try (Writer writer = Files.newBufferedWriter(outputPath, StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING)) {
215                     IOUtils.copy(mavenReaderFilter.filter(req), writer);
216                 }
217                 Files.setPosixFilePermissions(outputPath, POSIX_FILE_PERMISSIONS);
218             }
219         }
220     }
