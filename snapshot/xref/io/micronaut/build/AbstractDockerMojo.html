37  /**
38   * Abstract base class for mojos related to Docker files and builds.
39   *
40   * @author Álvaro Sánchez-Mariscal
41   * @author Iván López
42   * @since 1.1
43   */
44  public abstract class AbstractDockerMojo extends AbstractMojo {
45  
46      public static final String LATEST_TAG = "latest";
47      public static final String DEFAULT_BASE_IMAGE_GRAALVM_RUN = "frolvlad/alpine-glibc:alpine-3.12";
48      public static final String MOSTLY_STATIC_NATIVE_IMAGE_GRAALVM_FLAG = "-H:+StaticExecutableWithDynamicLibC";
49  
50      protected final MavenProject mavenProject;
51      protected final JibConfigurationService jibConfigurationService;
52      protected final ApplicationConfigurationService applicationConfigurationService;
53      protected final DockerService dockerService;
54  
55  
56      /**
57       * Additional arguments that will be passed to the <code>native-image</code> executable. Note that this will only
58       * be used when using a packaging of type <code>docker-native</code>. For <code>native-image</code> packaging
59       * you should use the
60       * <a href="https://www.graalvm.org/reference-manual/native-image/NativeImageMavenPlugin/#maven-plugin-customization">
61       * Native Image Maven Plugin
62       * </a> configuration options.
63       */
64      @Parameter(property = "micronaut.native-image.args")
65      protected List<String> nativeImageBuildArgs;
66  
67      /**
68       * List of additional arguments that will be passed to the application.
69       */
70      @Parameter(property = RunMojo.MN_APP_ARGS)
71      protected List<String> appArguments;
72  
73      /**
74       * The main class of the application, as defined in the
75       * <a href="https://www.mojohaus.org/exec-maven-plugin/java-mojo.html#mainClass">Exec Maven Plugin</a>.
76       */
77      @Parameter(defaultValue = RunMojo.EXEC_MAIN_CLASS, required = true)
78      protected String mainClass;
79  
80      /**
81       * Whether to produce a static native image when using <code>docker-native</code> packaging
82       */
83      @Parameter(defaultValue = "false", property = "micronaut.native-image.static")
84      protected Boolean staticNativeImage;
85  
86      /**
87       * The target runtime of the application
88       */
89      @Parameter(property = MicronautRuntime.PROPERTY, defaultValue = "NONE")
90      protected String micronautRuntime;
91  
92      /**
93       * The Docker image used to run the native image
94       * @since 1.2
95       */
96      @Parameter(property = "micronaut.native-image.base-image-run", defaultValue = DEFAULT_BASE_IMAGE_GRAALVM_RUN)
97      protected String baseImageRun;
98  
99      protected AbstractDockerMojo(MavenProject mavenProject, JibConfigurationService jibConfigurationService, ApplicationConfigurationService applicationConfigurationService, DockerService dockerService) {
100         this.mavenProject = mavenProject;
101         this.jibConfigurationService = jibConfigurationService;
102         this.applicationConfigurationService = applicationConfigurationService;
103         this.dockerService = dockerService;
104     }
105 
106     protected ArtifactVersion javaVersion() {
107         return new DefaultArtifactVersion(Optional.ofNullable(mavenProject.getProperties().getProperty("maven.compiler.target")).orElse(System.getProperty("java.version")));
108     }
109 
110     protected String graalVmVersion() {
111         return mavenProject.getProperties().getProperty("graal.version");
112     }
113 
114     protected String graalVmJvmVersion() {
115         String graalVmJvmVersion = "java11";
116         if (javaVersion().getMajorVersion() >= 17) {
117             graalVmJvmVersion = "java17";
118         }
119         return graalVmJvmVersion;
120     }
121 
122     protected String getFrom() {
123         if (staticNativeImage) {
124             // For building a static native image we need a base image with tools (cc, make,...) already installed
125             return jibConfigurationService.getFromImage().orElse("ghcr.io/graalvm/graalvm-ce:ol8-" + graalVmJvmVersion() + "-" + graalVmVersion());
126         } else {
127             return jibConfigurationService.getFromImage().orElse("ghcr.io/graalvm/native-image:ol8-" + graalVmJvmVersion() + "-" + graalVmVersion());
128         }
129     }
130 
131     protected Set<String> getTags() {
132         Set<String> tags = new HashSet<>();
133         Optional<String> toImageOptional = jibConfigurationService.getToImage();
134         String imageName = mavenProject.getArtifactId();
135         if (toImageOptional.isPresent()) {
136             String toImage = toImageOptional.get();
137             if (toImage.contains(":")) {
138                 tags.add(toImage);
139                 imageName = toImageOptional.get().split(":")[0];
140             } else {
141                 tags.add(toImage + ":" + LATEST_TAG);
142                 imageName = toImage;
143             }
144         } else {
145             tags.add(imageName + ":" + LATEST_TAG);
146         }
147         for (String tag : jibConfigurationService.getTags()) {
148             if (LATEST_TAG.equals(tag) && tags.stream().anyMatch(t -> t.contains(LATEST_TAG))) {
149                 continue;
150             }
151             tags.add(String.format("%s:%s", imageName, tag));
152         }
153         return tags;
154     }
155 
156     protected String getPort() {
157         String port = applicationConfigurationService.getServerPort();
158         return "-1".equals(port) ? DEFAULT_PORT : port;
159     }
160 
161     @SuppressWarnings("ResultOfMethodCallIgnored")
162     protected void copyDependencies() throws IOException {
163         List<String> imageClasspathScopes = Arrays.asList(Artifact.SCOPE_COMPILE, Artifact.SCOPE_RUNTIME);
164         mavenProject.setArtifactFilter(artifact -> imageClasspathScopes.contains(artifact.getScope()));
165         File target = new File(mavenProject.getBuild().getDirectory(), "dependency");
166         if (!target.exists()) {
167             target.mkdirs();
168         }
169         for (Artifact dependency : mavenProject.getArtifacts()) {
170             Files.copy(dependency.getFile().toPath(), target.toPath().resolve(dependency.getFile().getName()), StandardCopyOption.REPLACE_EXISTING);
171         }
172     }
173 
174     protected String getCmd() {
175         return "CMD [" +
176                 appArguments.stream()
177                         .map(s -> "\"" + s + "\"")
178                         .collect(Collectors.joining(", ")) +
179                 "]";
180     }
181 
182     protected String getGraalVmBuildArgs() {
183         if (nativeImageBuildArgs != null && !nativeImageBuildArgs.isEmpty()) {
184             return String.join(" ", nativeImageBuildArgs);
185         } else {
186             return "";
187         }
188     }
189 
190 }
